#plotting CDF of nodes
counts,bin_edges =np.histogram(survive['nodes'],bins=5,density=True)
pdf = counts/(sum(counts))
print(pdf);
print(bin_edges)

#compute CDF
cdf = np.cumsum(pdf)
plt.title('CDF for patients survived more than 5 years')
